JAVASCRIPT FUNCTIONS

We create "functions" to reuse some common actions. The only thing we need to do is changing the value without changing any code. We have three types of functions in JavaScript. First one is named functions, second one is anonymous functions and the third one is immediately invoked functions. Here are some code examples:
          
            // Find the age of a person
            function findAge(name, birthYear) {

              let date = new Date();
              let currentYear = date.getFullYear(); 
              let age = currentYear - birthYear;
            
              let output = name + " is " + age + " years old!";
            
              console.log(output);
            
              return output; // EXIT here !
            }
            console.log( findAge("Jane", 1974) );
            console.log( findAge("John", 1941) );
            console.log( findAge("Sam", 1990) );
            

            //Return day of the week
            function dayOfTheWeek(index) {

              let days = ["Sunday","Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"];

              return days[index];
            }
            console.log( dayOfTheWeek(3) );
